Create `lib/strategic/queries.ts`. This file contains async helper functions used by Server Components (pages). These are NOT Server Actions — they are plain async functions that accept a Supabase client or create one internally.

DESIGN: Accept the supabase client as a parameter (not create one inside) so pages can share a single client instance per request. This matches the pattern used in lib/auth/ helpers.

```typescript
// lib/strategic/queries.ts
// Server-side query helpers for strategic planning pages.
// These are imported by Server Components — NOT Server Actions.
// Each function accepts a supabase SupabaseClient to share the request-scoped client.

import type { SupabaseClient } from '@supabase/supabase-js'

export const KPI_PAGE_SIZE = 20  // D-21: 20 rows per page

/**
 * Fetch paginated KPIs with embedded objective title and all readings.
 * Readings are fetched as an array — caller uses getLatestReading() to find the most recent.
 * Approach A per RESEARCH.md Q3: no DISTINCT ON view needed at prototype scale.
 */
export async function getKpisWithReadings(
  supabase: SupabaseClient,
  { page = 1 }: { page?: number } = {}
) {
  const { data, count, error } = await supabase
    .from('kpis')
    .select(
      `
      id,
      title,
      description,
      owner_id,
      baseline_value,
      target_value,
      unit_of_measure,
      reporting_frequency,
      objective_id,
      strategic_objectives ( id, title ),
      kpi_readings ( actual_value, reporting_period, recorded_at ),
      user_profiles!owner_id ( full_name )
      `,
      { count: 'exact' }
    )
    .order('created_at', { ascending: false })
    .range((page - 1) * KPI_PAGE_SIZE, page * KPI_PAGE_SIZE - 1)

  return { data: data ?? [], count: count ?? 0, error }
}

/**
 * Fetch all strategic objectives for the authenticated institution.
 * RLS enforces institution_id scoping — no manual filter needed.
 */
export async function getObjectives(supabase: SupabaseClient) {
  const { data, error } = await supabase
    .from('strategic_objectives')
    .select(
      `
      id,
      title,
      description,
      status,
      nds2_pillar,
      institutional_goal,
      owner_id,
      start_date,
      target_date,
      created_at,
      user_profiles!owner_id ( full_name )
      `
    )
    .order('created_at', { ascending: false })

  return { data: data ?? [], error }
}

/**
 * Select the most recent reading from an array of kpi_readings.
 * Sorted by recorded_at descending; returns null if array is empty.
 * This is the Approach A "latest reading" selection per RESEARCH.md Q3.
 */
export function getLatestReading(
  readings: { actual_value: number; reporting_period: string; recorded_at: string }[]
): { actual_value: number; reporting_period: string; recorded_at: string } | null {
  if (!readings || readings.length === 0) return null
  return [...readings].sort(
    (a, b) => new Date(b.recorded_at).getTime() - new Date(a.recorded_at).getTime()
  )[0]
}
```

NOTE on user_profiles!owner_id join: This uses Supabase foreign key relationship syntax to join on owner_id. If the foreign key relationship between kpis.owner_id and user_profiles.id is not in the generated types, the select string still works at runtime — Supabase PostgREST resolves it from the schema. If TypeScript complains about the type shape, cast the result or use `as any` only for the join field until types/supabase.ts is regenerated after supabase db push.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Server Action | Form values arrive from client; Zod safeParse is the entry validation |
| Server Action → Supabase | Data passes to DB after role check; RLS is the second enforcement layer |
| JWT claims → role check | app_metadata.active_role is server-controlled; user cannot spoof it |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-P03-01 | Elevation of Privilege | createObjective role check | mitigate | `['admin','ceo'].includes(activeRole)` check before any DB write; RLS insert policy is second layer |
| T-02-P03-02 | Elevation of Privilege | recordKpiReading owner check | mitigate | Explicit `kpi.owner_id === user.id \|\| activeRole === 'admin'` before insert; RLS kpi_readings_insert policy as second layer |
| T-02-P03-03 | Tampering | createObjective / createKpi mass assignment | mitigate | Only `parsed.data` from Zod safeParse reaches the Supabase insert — unknown fields stripped automatically |
| T-02-P03-04 | Spoofing | JWT claim role check | mitigate | `user.app_metadata` is set by Supabase service role during the JWT hook — users cannot write to app_metadata |
| T-02-P03-05 | Information Disclosure | Error messages in Server Actions | mitigate | Generic 'An unexpected error occurred' message returned to client; full error logged server-side with console.error only |
</threat_model>
